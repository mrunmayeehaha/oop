#include <iostream>
#include <list>
using namespace std;

int main() {
    list<int> stack;

    stack.push_front(10);
    stack.push_front(20);
    stack.push_front(30);
    
    for(auto x : stack)
    cout<<x<<"\n";
    cout << "Top element: " << stack.front() << endl;

    stack.pop_front();

    cout << "New top after pop: " << stack.front() << endl;

    return 0;
}
